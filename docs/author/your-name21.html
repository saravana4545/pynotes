# Timer Decorator

import time

def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"Execution time: {end - start:.4f} seconds") # .4f - It prints with four decimal places.
        return result
    return wrapper

# Dict & Optional

from typing import Dict, Optional

def get_value(data: Dict[str, int], key: str) -> Optional[int]:
    return data.get(key)

data = {'apple': 10, 'banana': 5}

print(get_value(data, 'apple'))

10

print(get_value(data, 'grapes'))
