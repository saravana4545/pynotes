#Calculates the sum of its digits.Prints the result.

def sum_of_digits(num):
    digit_sum = 0
    while num > 0:
        digit_sum += num % 10
        num //= 10
    return digit_sum

list = 4545
print(sum_of_digits(list))

def large_and_small_nums(num):
    max_num = max(num)
    min_num = min(num)
    return max_num, min_num

def encode_data(userid, email, expiry_date):
    data         = f"{userid}|{email}|{expiry_date}"

Basic-Json

import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1

print(me.toJSON())
