def generate_squares(n):
    """
    Generate a list of squares of numbers from 1 to n.

    Parameters:
    n (int): The upper limit of the range.

    Returns:
    list: A list of squares of numbers from 1 to n.
    """

    squares = [n ** 2 for n in range(1, n+1)]
    return squares
