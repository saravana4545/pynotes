#Checks if the number is a prime number.
#Prints "Prime" or "Not Prime.

def check_prime(num):
    if num % 2:
        return f"{num} It's not a prime number"
    else:
        return f"{num} It's a prime number"

for i in range(1,11):
    print(check_prime(i))
