#Calculates the sum of its digits.Prints the result.

#create method called sum_of_digits()

def sum_of_digits(num):
    digit_sum = 0
    while num > 0:
        digit_sum += num % 10
        num //= 10
    return digit_sum

#create a variable and put the value for num

list = 4545
print(sum_of_digits(list))

#Finds and prints the smallest and largest numbers in the list.

#create a method called large_and_small_nums()

def large_and_small_nums(num):
    max_num = max(num)
    min_num = min(num)
    return max_num, min_num

#create a variable and put the value for num

list = (10,25,45,50,24)
large_and_small_nums(list)

(50, 10)

def decorator(func):
    def wrapper():
        print("Before function call")
        func()
        print("After function call")
    return wrapper

import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1
