# Operator Precedence

# Example 1: Without parentheses

result = 10 + 2 * 3

print("Result:", result)

Result: 16

# Example 2: With parentheses

result = (10 + 2) * 3

print("Result:", result)

Result: 36

# Example 3: Exponentiation

result = 2 ** 3 ** 2

print("Result:", result)

Result: 512

Score: 10
