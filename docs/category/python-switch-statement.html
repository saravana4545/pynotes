# Switch Using Dictionary format

days_dict = {
    1: "Monday",
    2: "Tuesday",
    3: "Wednesday",
    4: "Thursday",
    5: "Friday",
    6: "Saturday",
    7: "Sunday"
}

def get_day(day):
    return days_dict.get(day, "Invalid day")

print(get_day(5))

Friday

print(get_day(10))

Invalid day
