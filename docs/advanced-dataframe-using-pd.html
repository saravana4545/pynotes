#import pandas library

#!pip install pandas

import pandas as pd

#create a variable called data and put some values

data1 = {
    'Category': ['A', 'A', 'B', 'B'],
    'Type': ['X', 'Y', 'X', 'Y'],
    'Value': [10, 20, 30, 40]
}

df = pd.DataFrame(data1)

# Setting a multi-level index

df = df.set_index(['Category', 'Type'])
print(df)

               Value
Category Type       
A        X        10
         Y        20
B        X        30
         Y        40

# Accessing data using multi-level indexing

print(df.loc['A', 'X'])

Value    10
Name: (A, X), dtype: int64





#create pivot  table 

data2 = {
    'City': ['New York', 'Los Angeles', 'Chicago', 'New York'],
    'Year': [2020, 2020, 2021, 2021],
    'Sales': [100, 200, 150, 300]
}

df = pd.DataFrame(data2)

# Creating a pivot table

pivot = df.pivot_table(values='Sales', index='City', columns='Year', aggfunc='sum', fill_value=0)
print(pivot)

Year         2020  2021
City                   
Chicago         0   150
Los Angeles   200     0
New York      100   300





#Exploding Lists into Rows

data3 = {
    'Name': ['Alice', 'Bob'],
    'Hobbies': [['Reading', 'Cycling'], ['Swimming']]
}

df = pd.DataFrame(data3)

# Exploding lists into separate rows

df = df.explode('Hobbies')
print(df)

    Name   Hobbies
0  Alice   Reading
0  Alice   Cycling
1    Bob  Swimming
