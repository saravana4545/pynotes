```python

```


```python
pivot = pd.pivot_table(
    df, 
    values     ='Values', 
    index      ='Category', 
    columns    ='Subcategory', 
    aggfunc    ='sum', 
    fill_value =0
)
```


```python
print("Pivot Table:")
print(pivot)
```
