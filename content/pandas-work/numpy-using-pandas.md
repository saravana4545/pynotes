```python
# Handling Missing Data
```


```python
#import neccessary libraries
```


```python
import pandas as pd
```


```python
import numpy as np
```


```python
# Create a DataFrame with missing values
```


```python
#np.nan: Represents missing values.
```


```python

```


```python
data = {
    'Name': ['Alice', 'Bob', None],
    'Age': [25, np.nan, 35],
    'City': ['New York', None, 'Chicago']
}
```


```python
df = pd.DataFrame(data)
```


```python
# Fill missing values with 'Unknown' or 0
```


```python
#df.fillna(): Replaces missing values with specified values.
```


```python
df_filled = df.fillna({'Name': 'Unknown', 'Age': 0, 'City': 'Unknown'})
```


```python
#print the variable
```


```python
print(df_filled)
```

          Name   Age      City
    0    Alice  25.0  New York
    1      Bob   0.0   Unknown
    2  Unknown  35.0   Chicago

```python
# Merging DataFrames
```


```python
df1 = pd.DataFrame({'ID': [1, 2, 3], 'Name': ['Alice', 'Bob', 'Charlie']})
```


```python
df2 = pd.DataFrame({'ID': [1, 2, 4], 'Score': [85, 90, 95]})
```


```python
# Merge on the 'ID' column
```


```python
merged_df = pd.merge(df1, df2, on='ID', how='inner')
```


```python
print(merged_df)
```

       ID   Name  Score
    0   1  Alice     85
    1   2    Bob     90
