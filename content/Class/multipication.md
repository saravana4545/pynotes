```python
class Calculate:

    def __init__(self, num1, num2, num3):
        self.num1 = num1
        self.num2 = num2
        self.num3 = num3

    def multyply(self,num1, num2):
        mult = num1 * num2
        print(f"multy:{mult}")
        return mult
    
    def multyply_2(self, num2, num3):
        mult_2 = num2 * num3
        print(f"multply second num: {mult_2}")
        return mult_2
```

```python
class Calculate_2(Calculate):
    def __init__(self, num1, num2, num3):
        super().__init__(num1, num2, num3)
```
