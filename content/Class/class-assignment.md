class Parent1:
    def __init__(self, a, b, c):
        self.a = a
        self.b = b
        self.c = c

    def add(self, a, b, c):
        a1 = self.a + self.b
        a2 = self.b + self.c
        print(f"{a} and {b} is {a1}")
        print(f"{b} and {c} is {a2}")
        pass
        return a1, a2

    def sub(self, a, b):
        s1 = a - b
        s2 = self.add(a,b,c)
        print(f"{a} and {b} is {s1}")
        print(f"a1 and a2 is {s2}")
        return s1, s2
```

class Parent2(Parent1):
    def multi(self, s1, s2, a1, a2):
        m1 = super().sub(s1 * s2)
        m2 = super().add(a1 * a2)
        print(f"s1 and s2 is {m1}")
        print(f"a1 and a2 is {m2}")
        pass
        return m1, m2

    def div(self, m1, m2):
        d1 = m1 / m2
        print(f"m1 by m2 is {d1}")
        pass
        return d1
```
