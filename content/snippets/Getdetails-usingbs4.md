    books = soup.find_all('article', class_='product_pod')

        # Extract book link (relative URL)
        link = book.h3.a['href']
        full_link = f"https://books.toscrape.com/catalogue/{link}"
        
        print(f"Title: {title}")
        print(f"Price: {price}")
        print(f"Link: {full_link}")
